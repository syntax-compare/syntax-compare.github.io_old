4.1. for-loop (collection)
const fruits = ["apple", "banana", "mango"];

// option 1 (for...of)
for (const fruit of fruits) {
  console.log(fruit);
}

// option 2 (forEach)
fruits.forEach((fruit) => console.log(fruit));

> "apple"
> "banana"
> "mango"
